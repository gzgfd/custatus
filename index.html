    const COVERAGE_TZ = { label: "EST", gmtOffsetHours: -5 };


    // Coverage windows in EST (GMT-5): day 0=Sun ... 6=Sat

    const COVERAGE_SCHEDULE = [

      { day: 1, start: "12:00", end: "17:00" }, // Mon

      { day: 2, start: "12:00", end: "17:00" }, // Tue

      { day: 3, start: "08:00", end: "15:00" }, // Wed

      { day: 4, start: "12:00", end: "17:00" }, // Thu

    ];


    const pad2 = (n) => String(n).padStart(2, "0");

    const parseHHMM = (hhmm) => {

      const [h, m] = hhmm.split(":").map(Number);

      return { h, m: isNaN(m) ? 0 : m };

    };

    const minutesSinceMidnight = (date) => date.getHours() * 60 + date.getMinutes();


    function dateInGmtOffset(nowLocal, gmtOffsetHours){

      const utcMillis = nowLocal.getTime() + nowLocal.getTimezoneOffset() * 60 * 1000;

      return new Date(utcMillis + gmtOffsetHours * 60 * 60 * 1000);

    }


    function formatTime(date){

      const h = date.getHours();

      const m = date.getMinutes();

      const ampm = h >= 12 ? "PM" : "AM";

      const hr12 = ((h + 11) % 12) + 1;

      return `${hr12}:${pad2(m)} ${ampm}`;

    }


    function coverageStatus(nowLocal){

      const nowCoverage = dateInGmtOffset(nowLocal, COVERAGE_TZ.gmtOffsetHours);

      const dow = nowCoverage.getDay();

      const mins = minutesSinceMidnight(nowCoverage);


      const todays = COVERAGE_SCHEDULE.filter(s => s.day === dow);

      if (!todays.length) return { online:false, nowCoverage };


      for (const w of todays){

        const start = parseHHMM(w.start);

        const end = parseHHMM(w.end);

        const startM = start.h * 60 + start.m;

        const endM = end.h * 60 + end.m;

        if (mins >= startM && mins < endM){

          return { online:true, nowCoverage };

        }

      }

      return { online:false, nowCoverage };

    }

    const coverageDotEl = document.getElementById("coverageDot");

    function setDot(dotEl, state){

      dotEl.classList.remove("online","away","offline");

      dotEl.classList.add(state);

    }

    function setStatusText(textEl, state, label){

      textEl.classList.remove("online","away","offline");

      textEl.classList.add(state);

      textEl.textContent = label;

    }


    function tick(){

      const now = new Date();

      const cov = coverageStatus(now);


      // Header indicator (coverage window status)

      coverageDotEl.style.background = cov.online ? "var(--green)" : "var(--gray)";

      coverageDotEl.style.boxShadow = cov.online ? "0 0 0 3px rgba(34,197,94,.18)" : "0 0 0 3px rgba(156,163,175,.22)";

      coverageNowEl.textContent = ` • Now: ${formatTime(cov.nowCoverage)} ${COVERAGE_TZ.label}`;


      // Everyone’s local time always updates.

      // Activity indicator is based on coverage window (shared schedule).

      let onlineCount = 0;

      for (const card of cards){

        const m = card._member;

        const local = dateInGmtOffset(now, m.gmtOffsetHours);

        card._refs.time.textContent = formatTime(local);


        if (cov.online){

          onlineCount++;

          setDot(card._refs.dot, "online");

          setStatusText(card._refs.status, "online", "Online");

        } else {

          setDot(card._refs.dot, "offline");

          setStatusText(card._refs.status, "offline", "Offline");

        }

      }


      onlineCountEl.textContent = `${onlineCount} of ${TEAM.length} online`;

    }
